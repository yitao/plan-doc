

## 创建索引

`PUT /{索引名称}`

```json
{
  "settings": {
    "index": {
} 
},
	"mappings": {
		"{type_name}": {
			"properties": {
					"{字符串格式字段名称}": {
						"type": "text",
						"fielddata": true,
						"analyzer": "ik",
						"fields": {
							"keyword": {
								"type": "keyword"
							}
						}
					}
				,
				
					"{日期格式字段名称}": {
						"type": "date",
						"format": "yyyy-MM-dd HH:mm:ss|yyyy-MM-dd|epoch_millis"
					},
				
				
					"{内嵌对象格式字段名称}": {
						"type": "nested",
						"properties": [{
							"{内嵌对象字段名称1}": {
								"type": "keyword"
							},
							"{内嵌对象字段名称2}": {
								"type": "long"
							}
						}]
					}
				
			}
		}
	}
}
```

    



## 删除索引

`DELETE /{索引名称}`


## 重新索引

`POST /_reindex`

```json
{
  "source": {
    "index": "{原始index名称}"
  },
  "dest": {
    "index": "{新index名称}"
  }
}
```
